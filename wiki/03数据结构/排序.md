冒泡排序

```python
# python 冒泡排序
def maopao_sort(arr):
    n = len(arr)
    # 遍历所有数组元素
    for i in range(n):
        for j in range(n-i-1):
            if arr[j] > arr[j+1] :                  # 左边元素>右边元素
                arr[j], arr[j+1] = arr[j+1], arr[j] # 交换

arr = [64, 34, 25, 12, 22, 11, 90]
maopao_sort(arr)
for i in range(len(arr)): # 打印正序列表
    print (arr[i])
```
